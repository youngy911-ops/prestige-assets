```typescript
// Add above the POST function in route.ts
// Parse 'key: value\n' lines from inspection_notes into a Record<string,string>
// Excludes the 'Notes' freeform key; handles null input
function parseStructuredFields(notes: string | null): Record<string, string> {
  if (!notes) return {}
  const result: Record<string, string> = {}
  for (const line of notes.split('\n')) {
    const colonIdx = line.indexOf(': ')
    if (colonIdx === -1) continue
    const key = line.slice(0, colonIdx).trim()
    const value = line.slice(colonIdx + 2).trim()
    // 'Notes' is the freeform textarea key — not a structured field
    if (key === 'Notes' || !key || !value) continue
    result[key] = value
  }
  return result
}
```
