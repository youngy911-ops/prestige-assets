```typescript
{ key: 'vin', ..., aiExtractable: true,
  aiHint: 'Build plate or compliance plate: 17-character alphanumeric. Also stamped on chassis rail. Never infer — only extract if visible.' },

{ key: 'make', ..., aiExtractable: true,
  aiHint: 'Build plate: manufacturer name (e.g. MACK, Kenworth, Volvo, Isuzu, Hino, Mercedes-Benz, DAF).' },

{ key: 'model', ..., aiExtractable: true,
  aiHint: 'Build plate: model designation (e.g. Granite, T908, FH16, NLR). Read exactly as printed.' },

{ key: 'year', ..., aiExtractable: true,
  aiHint: 'Build plate: "Year of Manufacture" or "DOM" (Date of Manufacture). 4-digit year only.' },

{ key: 'cab_type', ..., aiExtractable: true,
  aiHint: 'Visual from exterior photos. Common values: Day Cab, Sleeper, Crew Cab. Infer from shape if not labelled.' },

{ key: 'gearbox_make', ..., aiExtractable: true,
  aiHint: 'Badge on gearbox or infer from make/model (e.g. MACK mDrive, Volvo I-Shift, Eaton Fuller, ZF, Allison).' },

{ key: 'drive_type', ..., aiExtractable: true,
  aiHint: 'Build plate or infer from model. Format: 6x4, 4x2, 8x4, 6x2. Count driven axles from under-vehicle photos.' },

{ key: 'engine_manufacturer', ..., aiExtractable: true,
  aiHint: 'Engine badge or build plate. Infer from make/model if not visible (e.g. MACK→MACK/MP8, Kenworth→Cummins/PACCAR).' },

{ key: 'engine_series', ..., aiExtractable: true,
  aiHint: 'Badge or infer from make/model/year (e.g. MP8, MP10, ISX15, X15, D13, DD15, MX-13, C15). Model-specific constant.' },

{ key: 'engine_type', ..., aiExtractable: true,
  aiHint: 'Infer from make/model: typically "Turbodiesel Inline-6". Use engine family name if known (e.g. MP8 Turbodiesel).' },

{ key: 'engine_size', ..., aiExtractable: true,
  aiHint: 'Build plate or badge. Format: displacement in litres with one decimal (e.g. 12.7L, 15.0L, 13.0L). Infer from model.' },

{ key: 'fuel_type', ..., aiExtractable: true,
  aiHint: 'Heavy trucks are almost always Diesel. Only return non-Diesel if there is clear evidence (e.g. LPG badge, electric motor).' },

{ key: 'variant', ..., aiExtractable: true,
  aiHint: 'Build plate variant or series field (e.g. GU500, T404ST, FH540). Read exactly as printed. Null if not visible.' },

{ key: 'compliance_date', ..., aiExtractable: true,
  aiHint: 'Compliance plate (silver/gold plate inside door or under bonnet). Format MM/YYYY. e.g. 03/2012.' },

{ key: 'odometer', ..., aiExtractable: true, inspectionPriority: true,
  aiHint: 'Instrument cluster display. Digits only, no units. Only extract if clearly readable — do NOT guess.' },

{ key: 'registration_number', ..., aiExtractable: true, inspectionPriority: true,
  aiHint: 'Registration plate on front or rear of vehicle. Australian format varies by state. Null if no plate visible.' },

{ key: 'colour', ..., aiExtractable: true,
  aiHint: 'Exterior cab colour from photos. Use common colour names (White, Red, Blue, Yellow, Orange, Green, Silver, Black).' },

{ key: 'body_type', ..., aiExtractable: true,
  aiHint: 'Body fitted to chassis from exterior photos (e.g. Tipper, Flat Deck, Curtainsider, Tanker, Crane, Service). Null if prime mover only.' },

{ key: 'hourmeter', ..., aiExtractable: true, inspectionPriority: true,
  aiHint: 'Instrument cluster hour meter display. Digits only. Only extract if clearly readable — do NOT guess.' },

{ key: 'tyre_size', ..., aiExtractable: true,
  aiHint: 'Tyre sidewall markings. Format: width/aspect R diameter (e.g. 315/80R22.5, 11R22.5). Use steer tyres as reference.' },

{ key: 'transmission', ..., aiExtractable: true,
  aiHint: 'Infer from make/model/year (e.g. Automated Manual, Manual, Automatic). Common: mDrive, I-Shift, Eaton UltraShift, Allison.' },

{ key: 'suspension', ..., aiExtractable: true,
  aiHint: 'Infer from model or visible from under-vehicle photos. Common: Air, Leaf Spring, Airbag, Hendrickson.' },

{ key: 'axle_configuration', ..., aiExtractable: true,
  aiHint: 'Build plate or infer from visual. Format: number of axles × driven (e.g. 6x4, 4x2, 8x4). Same as drive_type.' },

{ key: 'brakes', ..., aiExtractable: true,

{ key: 'gcm', ..., aiExtractable: true,

{ key: 'gvm', ..., aiExtractable: true,
```
