    ```tsx
    import { CheckCircle2, AlertCircle, MinusCircle } from 'lucide-react'

      level: ConfidenceLevel
      className?: string
    }

    const CONFIG: Record<ConfidenceLevel, {
      icon: React.ComponentType<{ className?: string }>
      label: string
      className: string
    }> = {
      high: {
        icon: CheckCircle2,
        label: 'High',
        className: 'text-green-400 bg-transparent border-green-400/30',
      },
      medium: {
        icon: AlertCircle,
        label: 'Medium',
        className: 'text-amber-400 bg-transparent border-amber-400/30',
      },
      low: {
        icon: MinusCircle,
        label: 'Low',
        className: 'text-white/40 bg-transparent border-white/20',
      },
      not_found: {
        icon: MinusCircle,
        label: 'Not found',
        className: 'text-white/40 bg-transparent border-white/20',
      },
    }

    export function ConfidenceBadge({ level, className }: ConfidenceBadgeProps) {
      const { icon: Icon, label, className: levelClass } = CONFIG[level]
      return (
        <Badge
          variant="outline"
          className={cn('gap-1 text-xs font-normal px-1.5 py-0.5', levelClass, className)}
        >
          <Icon className="w-3 h-3 flex-shrink-0" aria-hidden="true" />
          <span className="sr-only">{level === 'not_found' ? 'Not found' : `${label} confidence`}</span>
          <span aria-hidden="true">{label}</span>
        </Badge>
      )
    }
    ```

    ```tsx
    'use client'

    import { useRef, useState } from 'react'
    import { Input } from '@/components/ui/input'
    import { Label } from '@/components/ui/label'
    import { Card, CardContent } from '@/components/ui/card'
    import { saveInspectionNotes } from '@/lib/actions/inspection.actions'
    import type { FieldDefinition } from '@/lib/schema-registry/types'

    interface InspectionNotesSectionProps {
      assetId: string
      priorityFields: FieldDefinition[]   // from getInspectionPriorityFields(assetType)
      initialNotes: string                // from assets.inspection_notes (empty string if null)
    }

    // Placeholder copy per known field keys
    const FIELD_PLACEHOLDER: Record<string, string> = {
      odometer: 'e.g. 187,450',
      hourmeter: 'e.g. 4,200',
      hours: 'e.g. 4,200',
      registration_number: 'e.g. 123ABC',
      registration_expiry: 'e.g. 01/2026',
      registration: 'e.g. 123ABC',
      service_history: 'e.g. Full dealer history',
      tare: 'e.g. 4,500',
      atm: 'e.g. 12,000',
      pin: 'e.g. 1FT8W3DT2NED12345',
      serial: 'e.g. CAT00329X00123',
      vin: 'e.g. 1HGCM82633A123456',
      max_lift_capacity: 'e.g. 2,500 kg',
    }

    export function InspectionNotesSection({
      assetId,
      priorityFields,
      initialNotes,
    }: InspectionNotesSectionProps) {
      const [notes, setNotes] = useState(initialNotes)
      const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null)

      function handleNotesChange(value: string) {
        setNotes(value)
        if (debounceRef.current) clearTimeout(debounceRef.current)
        debounceRef.current = setTimeout(() => {
          saveInspectionNotes(assetId, value)
        }, 500)
      }

      return (
        <Card className="bg-[var(--card)] border-white/10">
          <CardContent className="pt-4 pb-4 px-4 space-y-3">
            <h2 className="text-sm font-semibold text-white">Inspection Notes</h2>

            {/* Structured priority fields — one input per field */}
            {priorityFields.map((field) => (
              <div key={field.key} className="space-y-1">
                <Label
                  htmlFor={`inspection-${field.key}`}
                  className="text-xs text-white/65"
                >
                  {field.label}
                </Label>
                <Input
                  id={`inspection-${field.key}`}
                  className="h-9 text-sm bg-transparent border-white/20 text-white placeholder:text-white/40 focus:border-[oklch(0.29_0.07_248)]"
                  placeholder={FIELD_PLACEHOLDER[field.key] ?? 'Enter value'}
                  onChange={(e) => {
                    // Each structured field saves as part of the freeform notes
                    // Format: "Field Label: value" appended to notes text
                    // This is a simplified approach — structured fields feed into the
                    // freeform notes textarea as context for the AI
                    handleNotesChange(e.target.value)
                  }}
                />
              </div>
            ))}

            {/* Freeform notes textarea */}
            <div className="space-y-1">
              <Label
                htmlFor="inspection-other-notes"
                className="text-xs text-white/65"
              >
                Other notes
              </Label>
              <textarea
                id="inspection-other-notes"
                value={notes}
                onChange={(e) => handleNotesChange(e.target.value)}
                className="w-full min-h-[80px] rounded-md border border-white/20 bg-transparent px-3 py-2 text-sm text-white placeholder:text-white/40 focus:outline-none focus:border-[oklch(0.29_0.07_248)] resize-none"
                placeholder="VIN, rego, dimensions, body builder, service history, number of keys, condition notes…"
              />
            </div>
          </CardContent>
        </Card>
      )
    }
    ```

    NOTE on structured fields: The CONTEXT.md decision is that structured fields AND the freeform textarea are passed to GPT-4o. For v1, the structured fields should be clearly labeled in the notes so AI sees "Odometer: 187,450" etc. Revise the onChange for structured fields to build a combined string if needed — the key design principle is that ALL text from the notes section reaches the AI. Discuss with Jack in Phase 5 accuracy validation if the concatenation approach needs refinement.

    **1. src/components/asset/ExtractionTriggerState.tsx**

    Client Component — shows "Run AI Extraction" CTA + "Skip to Manual Entry" link:

    ```tsx
    'use client'

    import { useState } from 'react'
    import { useRouter } from 'next/navigation'
    import Link from 'next/link'

    interface ExtractionTriggerStateProps {
      assetId: string
      hasPhotos: boolean
    }

    export function ExtractionTriggerState({ assetId, hasPhotos }: ExtractionTriggerStateProps) {
      const [isStarting, setIsStarting] = useState(false)
      const router = useRouter()

      async function handleRunExtraction() {
        if (!hasPhotos) return
        setIsStarting(true)
        try {
          await fetch('/api/extract', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ assetId }),
          })
          // Navigate immediately — extract page reads state from DB
          router.push(`/assets/${assetId}/extract`)
          router.refresh()
        } catch {
          setIsStarting(false)
        }
      }

      if (!hasPhotos) {
        return (
          <div className="space-y-3">
            <div className="rounded-md border border-white/10 bg-[var(--card)] p-4 text-center">
              <p className="text-sm font-semibold text-white mb-1">No photos uploaded</p>
              <p className="text-sm text-white/65">
                Upload photos first, then run AI extraction. Or skip to manual entry.
              </p>
            </div>
            <Link
              href={`/assets/${assetId}/review`}
              className="flex items-center justify-center w-full h-11 rounded-md bg-[oklch(0.29_0.07_248)] hover:bg-[oklch(0.29_0.07_248)]/90 text-white font-medium text-sm transition-colors"
            >
              Skip to Manual Entry
            </Link>
          </div>
        )
      }

      return (
        <div className="space-y-2">
          <button
            onClick={handleRunExtraction}
            disabled={isStarting}
            className="flex items-center justify-center w-full h-11 rounded-md bg-[oklch(0.29_0.07_248)] hover:bg-[oklch(0.29_0.07_248)]/90 text-white font-medium text-sm transition-colors disabled:opacity-50"
          >
            {isStarting ? 'Starting extraction…' : 'Run AI Extraction'}
          </button>
          <div className="text-center">
            <Link
              href={`/assets/${assetId}/review`}
              className="text-sm text-white/65 hover:text-white transition-colors"
            >
              Skip to Manual Entry
            </Link>
          </div>
        </div>
      )
    }
    ```

    **5. src/app/(app)/assets/[id]/extract/page.tsx**

    Server Component — reads DB state, dispatches to correct state component:

    ```tsx
    import { redirect } from 'next/navigation'
    import { ChevronLeft } from 'lucide-react'
    import Link from 'next/link'
    import { createClient } from '@/lib/supabase/server'
    import { getFieldsSortedBySfOrder } from '@/lib/schema-registry'
    import type { AssetType } from '@/lib/schema-registry/types'
    import { ExtractionTriggerState } from '@/components/asset/ExtractionTriggerState'
    import { ExtractionResultPanel } from '@/components/asset/ExtractionResultPanel'
    import type { ExtractionResult } from '@/lib/ai/extraction-schema'

    interface ExtractPageProps {
      params: Promise<{ id: string }>
    }

    export default async function ExtractPage({ params }: ExtractPageProps) {
      const { id: assetId } = await params
      const supabase = await createClient()

      // Auth check
      const { data: { user } } = await supabase.auth.getUser()
      if (!user) redirect('/login')

      // Load asset with extraction_result
      const { data: asset } = await supabase
        .from('assets')
        .select('id, asset_type, asset_subtype, extraction_result')
        .eq('id', assetId)
        .single()

      if (!asset) redirect('/assets/new')

      // Load photos count to determine if photos exist
      const { count: photoCount } = await supabase
        .from('asset_photos')
        .select('id', { count: 'exact', head: true })
        .eq('asset_id', assetId)

      const hasPhotos = (photoCount ?? 0) > 0
      const hasExtractionResult = asset.extraction_result != null

      // All fields for the asset type sorted by sfOrder (used in result panel)
      const allFields = getFieldsSortedBySfOrder(asset.asset_type as AssetType)

      return (
        <div className="max-w-[480px] mx-auto px-4 pt-8 pb-[calc(env(safe-area-inset-bottom)+80px)]">
          {/* Header */}
          <div className="flex items-center gap-3 mb-6">
            <Link
              href={`/assets/${assetId}/photos`}
              className="text-white/65 hover:text-white transition-colors p-1 -ml-1"
              aria-label="Back"
            >
              <ChevronLeft className="w-5 h-5" />
            </Link>
            <div>
              <h1 className="text-xl font-semibold text-white">AI Extraction</h1>
              <p className="text-sm text-white/65">
                {asset.asset_subtype
                  ? `${asset.asset_type} — ${asset.asset_subtype}`
                  : asset.asset_type}
              </p>
            </div>
          </div>

          {/* State dispatcher */}
          {hasExtractionResult ? (
            <ExtractionResultPanel
              assetId={assetId}
              fields={allFields}
              extractionResult={asset.extraction_result as ExtractionResult}
            />
          ) : (
            <ExtractionTriggerState
              assetId={assetId}
              hasPhotos={hasPhotos}
            />
          )}
        </div>
      )
    }
    ```
